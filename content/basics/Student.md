```python
# Define grade calculator
def grade(avg):
    if avg >= 90:
        return "A+"
    elif avg >= 80:
        return "A"
    elif avg >= 70:
        return "B"
    elif avg >= 60:
        return "C"
    else:
        return "F"

```


```python
# Apply grade
df["Grade"] = df["Average"].apply(grade)
df

```
