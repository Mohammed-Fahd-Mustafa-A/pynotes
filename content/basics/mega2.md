# Topic: Sorting Algorithms
def bubble_sort(arr):
 for i in range(len(arr)):
  for j in range(0, len(arr)-i-1):
   if arr[j]>arr[j+1]: arr[j], arr[j+1]=arr[j+1], arr[j]
a = [5,3,8,6]
bubble_sort(a)
print(a)
