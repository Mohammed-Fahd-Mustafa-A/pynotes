# 6. Add a column for rating category
def rate_group(r):
    if r >= 8.5:
        return "Excellent"
    elif r >= 8.0:
        return "Good"
    elif r >= 7.5:
        return "Average"
    else:
        return "Below Average"

df["rating_group"] = df["rating"].apply(rate_group)
