def factorial(n):
    return 1 if n==0 else n * factorial(n-1)
print(factorial(5))

    def wrapper():
        print('Before')
        fn()
        print('After')
    return wrapper
@decorator
def say(): print('Hi')

def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)
print(fib(5))
